9real_T look1_binlxpw(real_T u0, const real_T bp0[], const real_T table[],



10                     uint32_T maxIndex)



11{



12  real_T frac;



13  real_T yL_0d0;



14  uint32_T bpIdx;



15  uint32_T iLeft;



16  uint32_T iRght;



17  if (u0 <= bp0[0U]) {



18    iLeft = 0U;



19    frac = (u0 - bp0[0U]) / (bp0[1U] - bp0[0U]);



20  } else if (u0 < bp0[maxIndex]) {



21    bpIdx = maxIndex >> 1U;



22    iLeft = 0U;



23    iRght = maxIndex;



24    while (iRght - iLeft > 1U) {



25      if (u0 < bp0[bpIdx]) {



26        iRght = bpIdx;



27      } else {



28        iLeft = bpIdx;



29      }



30



31      bpIdx = (iRght + iLeft) >> 1U;



32    }



33



34    frac = (u0 - bp0[iLeft]) / (bp0[iLeft + 1U] - bp0[iLeft]);



35  } else {



36    iLeft = maxIndex - 1U;



37    frac = (u0 - bp0[maxIndex - 1U]) / (bp0[maxIndex] - bp0[maxIndex - 1U]);



38  }



39



40  yL_0d0 = table[iLeft];



41  return (table[iLeft + 1U] - yL_0d0) * frac + yL_0d0;



42}

44void Accumulator_step(void)



45{



46  real_T rtb_Rsz[126];



47  real_T rtb_Switch[126];



48  real_T a;



49  real_T rtb_Divide;



50  real_T rtb_Product1;



51  real_T rtb_Rsz_m;



52  real_T rtb_tauz;



53  int32_T i;



54  rtb_Product1 = Accumulator_P.Ts_sim * Accumulator_U.Current;



55  for (i = 0; i < 126; i++) {



56    rtb_Divide = Accumulator_DW.Memory_PreviousInput[i] /



57      Accumulator_DW.Memory2_PreviousInput[i];



58    rtb_Rsz_m = look1_binlxpw(rtb_Divide, Accumulator_P.soc, Accumulator_P.Rs,



59      12U);



60    rtb_tauz = look1_binlxpw(rtb_Divide, Accumulator_P.soc, Accumulator_P.tau,



61      12U);



62    a = rtb_Rsz_m * rtb_tauz;



63    rtb_Rsz_m = (rtb_Rsz_m + look1_binlxpw(rtb_Divide, Accumulator_P.soc,



64      Accumulator_P.Rp, 12U)) * Accumulator_P.Ts_sim;



65    rtb_Rsz_m = (((2.0 * a + rtb_Rsz_m) * Accumulator_U.Current + (rtb_Rsz_m -



66      2.0 * a) * Accumulator_DW.Memory3_PreviousInput) - (Accumulator_P.Ts_sim -



67      2.0 * rtb_tauz) * Accumulator_DW.Memory4_PreviousInput[i]) / (2.0 *



68      rtb_tauz + Accumulator_P.Ts_sim);



69    rtb_tauz = rtb_Rsz_m + look1_binlxpw(rtb_Divide, Accumulator_P.soc,



70      Accumulator_P.ocv, 12U);



71    Accumulator_Y.Voltages[i] = rtb_tauz;



72    Accumulator_Y.SOCs[i] = rtb_Divide;



73    Accumulator_Y.Capacities[i] = Accumulator_DW.Memory2_PreviousInput[i];



74    rtb_Rsz[i] = rtb_Rsz_m;



75    rtb_Switch[i] = (rtb_Product1 + Accumulator_DW.Memory_PreviousInput[i]) -



76      rtb_tauz / Accumulator_P.R_short * Accumulator_U.Balances[i];



77  }



78



79  rtb_Product1 = exp(Accumulator_P.Cap_det / (Accumulator_P.C_r * 3600.0) *



80                     Accumulator_DW.Memory1_PreviousInput);



81  Accumulator_DW.Memory3_PreviousInput = Accumulator_U.Current;



82  for (i = 0; i < 126; i++) {



83    rtb_tauz = rtb_Switch[i];



84    rtb_Divide = Accumulator_DW.Memory2_PreviousInput[i] * rtb_Product1;



85    Accumulator_DW.Memory4_PreviousInput[i] = rtb_Rsz[i];



86    if (rtb_tauz > rtb_Divide) {



87      Accumulator_DW.Memory_PreviousInput[i] = rtb_Divide;



88    } else if (rtb_tauz < Accumulator_P.Constant1_Value) {



89      Accumulator_DW.Memory_PreviousInput[i] = Accumulator_P.Constant1_Value;



90    } else {



91      Accumulator_DW.Memory_PreviousInput[i] = rtb_tauz;



92    }



93



94    Accumulator_DW.Memory2_PreviousInput[i] = rtb_Divide;



95  }



96



97  Accumulator_DW.Memory1_PreviousInput = fabs(Accumulator_U.Current +



98    Accumulator_DW.Memory1_PreviousInput);



99  rt_UpdateTXYLogVars(Accumulator_M->rtwLogInfo,



100                      (&Accumulator_M->Timing.taskTime0));



101



102  {



103    if ((rtmGetTFinal(Accumulator_M)!=-1) &&



104        !((rtmGetTFinal(Accumulator_M)-Accumulator_M->Timing.taskTime0) >



105          Accumulator_M->Timing.taskTime0 * (DBL_EPSILON))) {



106      rtmSetErrorStatus(Accumulator_M, "Simulation finished");



107    }



108  }



109



110  if (!(++Accumulator_M->Timing.clockTick0)) {



111    ++Accumulator_M->Timing.clockTickH0;



112  }



113



114  Accumulator_M->Timing.taskTime0 = Accumulator_M->Timing.clockTick0 *



115    Accumulator_M->Timing.stepSize0 + Accumulator_M->Timing.clockTickH0 *



116    Accumulator_M->Timing.stepSize0 * 4294967296.0;



117}
