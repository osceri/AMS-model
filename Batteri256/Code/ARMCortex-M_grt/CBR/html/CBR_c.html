21real_T look1_binlxpw(real_T u0, const real_T bp0[], const real_T table[],



22                     uint32_T maxIndex)



23{



24  real_T frac;



25  real_T yL_0d0;



26  uint32_T bpIdx;



27  uint32_T iLeft;



28  uint32_T iRght;



29  if (u0 <= bp0[0U]) {



30    iLeft = 0U;



31    frac = (u0 - bp0[0U]) / (bp0[1U] - bp0[0U]);



32  } else if (u0 < bp0[maxIndex]) {



33    bpIdx = maxIndex >> 1U;



34    iLeft = 0U;



35    iRght = maxIndex;



36    while (iRght - iLeft > 1U) {



37      if (u0 < bp0[bpIdx]) {



38        iRght = bpIdx;



39      } else {



40        iLeft = bpIdx;



41      }



42



43      bpIdx = (iRght + iLeft) >> 1U;



44    }



45



46    frac = (u0 - bp0[iLeft]) / (bp0[iLeft + 1U] - bp0[iLeft]);



47  } else {



48    iLeft = maxIndex - 1U;



49    frac = (u0 - bp0[maxIndex - 1U]) / (bp0[maxIndex] - bp0[maxIndex - 1U]);



50  }



51



52  yL_0d0 = table[iLeft];



53  return (table[iLeft + 1U] - yL_0d0) * frac + yL_0d0;



54}

70                      np, int32_T nq, int32_T iwork[126], real_T xwork[126])



71{

74  int32_T n;

80      iout = offset + q;



81      iwork[q] = idx[iout];

83    }

85    n = 0;

101        x[iout] = xwork[q];



102        if (q + 1 < n_tmp) {



103          q++;



104        } else {



105          q = iout - n;

107            iout = (q + n) + 1;



108            idx[iout] = iwork[n];



109            x[iout] = xwork[n];



110            n++;



111          }

117  }

120static void CBR_merge_block(int32_T idx[126], real_T x[126], int32_T offset,



121  int32_T n, int32_T preSortLevel, int32_T iwork[126], real_T xwork[126])



122{



123  int32_T bLen;



124  int32_T nPairs;



125  int32_T nTail;



126  int32_T tailOffset;



127  nPairs = n >> preSortLevel;



128  bLen = 1 << preSortLevel;



129  while (nPairs > 1) {



130    if ((nPairs & 1U) != 0U) {



131      nPairs--;



132      tailOffset = bLen * nPairs;



133      nTail = n - tailOffset;



134      if (nTail > bLen) {



135        CBR_merge(idx, x, offset + tailOffset, bLen, nTail - bLen, iwork, xwork);



136      }



137    }



138



139    tailOffset = bLen << 1;



140    nPairs >>= 1;



141    for (nTail = 0; nTail < nPairs; nTail++) {



142      CBR_merge(idx, x, offset + nTail * tailOffset, bLen, bLen, iwork, xwork);



143    }



144



145    bLen = tailOffset;



146  }



147



148  if (n > bLen) {



149    CBR_merge(idx, x, offset, bLen, n - bLen, iwork, xwork);



150  }



151}

384    idx = 1;

426    rtb_tauz1) * CBR_DW.Memory5_PreviousInput) - (rtb_Rsz1 - 2.0 * a) *
